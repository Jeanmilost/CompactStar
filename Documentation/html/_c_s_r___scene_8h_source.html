122typedef struct
123{
124 void* m_pModel; // the model to draw
125 CSR_EModelType m_Type; // model type (a simple mesh, a model or a complex MDL model)
126 CSR_ECollisionType m_CollisionType; // collision type to apply to model
127 CSR_Array* m_pMatrixArray; // matrices sharing the same model, e.g. all the walls of a room
128 CSR_AABBNode* m_pAABBTree; // aligned-axis bounding box trees owned by the model
129 size_t m_AABBTreeCount; // aligned-axis bounding box tree count
130 size_t m_AABBTreeIndex; // aligned-axis bounding box tree index to use for the collision detection
131} CSR_SceneItem;
132
136typedef struct
137{
138 CSR_Color m_Color; // the scene background color
139 CSR_Matrix4 m_ProjectionMatrix; // the scene projection matrix
140 CSR_Matrix4 m_ViewMatrix; // the scene view matrix
141 CSR_Vector3 m_GroundDir; // the ground direction in the whole scene
142 CSR_Mesh* m_pSkybox; // skybox geometry (because there is only one skybox per scene)
143 CSR_SceneItem* m_pItem; // the items in this list will be drawn in the scene
144 size_t m_ItemCount; // number of items
145 CSR_SceneItem* m_pTransparentItem; // the items in this list will be drawn on the scene end, allowing transparency
146 size_t m_TransparentItemCount; // number of transparent items
147} CSR_Scene;

190typedef struct
191{

195} CSR_CollisionInput;

201{
