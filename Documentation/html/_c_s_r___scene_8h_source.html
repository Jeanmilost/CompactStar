124typedef struct
125{
126 void* m_pModel; // the model to draw
127 CSR_EModelType m_Type; // model type (a simple mesh, a model or a complex animated model)
128 CSR_ECollisionType m_CollisionType; // collision type to apply to model
129 CSR_Array* m_pMatrixArray; // matrices sharing the same model, e.g. all the walls of a room
130 CSR_Collider* m_pCollider; // collider for the GJK algorithm
131 CSR_AABBNode* m_pAABBTree; // aligned-axis bounding box trees owned by the model
132 size_t m_AABBTreeCount; // aligned-axis bounding box tree count
133 size_t m_AABBTreeIndex; // aligned-axis bounding box tree index to use for the collision detection
134} CSR_SceneItem;
135
139typedef struct
140{
141 CSR_Color m_Color; // the scene background color
142 CSR_Matrix4 m_ProjectionMatrix; // the scene projection matrix
143 CSR_Matrix4 m_ViewMatrix; // the scene view matrix
144 CSR_Vector3 m_GroundDir; // the ground direction in the whole scene
145 CSR_Mesh* m_pSkybox; // skybox geometry (because there is only one skybox per scene)
146 CSR_SceneItem* m_pItem; // the items in this list will be drawn in the scene
147 size_t m_ItemCount; // number of items
148 CSR_SceneItem* m_pTransparentItem; // the items in this list will be drawn on the scene end, allowing transparency
149 size_t m_TransparentItemCount; // number of transparent items
150} CSR_Scene;
151

179typedef struct
180{
181 void* m_pModel; // the hit model
182 CSR_EModelType m_Type; // model type (a simple mesh, a model or a complex MDL model)
183 CSR_Matrix4 m_Matrix; // model matrix
184 CSR_AABBNode* m_pAABBTree; // aligned-axis bounding box tree in which the collision was found
185 CSR_Polygon3Buffer m_Polygons; // hit polygons in the model
186} CSR_HitModel;
187
194typedef struct
195{
196 CSR_Ray3 m_MouseRay; // ray starting from the mouse position, transformed in the viewport coordinates system
197 CSR_Sphere m_BoundingSphere; // bounding sphere representing the model or point of view at its current position
198 CSR_Vector3 m_CheckPos; // the model or point of view position to check
199} CSR_CollisionInput;
200
204typedef struct
205{
206 CSR_ECollisionType m_Collision; // found collision type in the scene
207 float m_GroundPos; // the ground position on the y axis, M_CSR_NoGround if no ground was found
208 CSR_Plane m_CollisionPlane; // the collision plane, in case a collision was found
209 CSR_Plane m_GroundPlane; // the ground plane, in case a ground was found
210 CSR_Array* m_pHitModels; // models hit by the mouse ray
211 CSR_Array* m_pColliders; // pairs of colliding colliders, array should be read as follow:
212 // [0] collides [1], [2] collides [3], ...
213} CSR_CollisionOutput;
